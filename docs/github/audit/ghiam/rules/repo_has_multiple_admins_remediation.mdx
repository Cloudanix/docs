import boto3

# Define AWS credentials
aws_access_key_id = 'YOUR_AWS_ACCESS_KEY_ID'

# Define the maximum number of admins allowed per repository
max_admins = 5

# Connect to AWS CodeCommit using boto3

# Retrieve the list of all repositories
response = client.list_repositories()

# For each repository, retrieve the list of admins and remove excess admins
for repo in response['repositories']:
    repo_name = repo['repositoryName']
    admins = client.list_repository_administrators(repositoryName=repo_name)['administrators']

        excess_admins = admins[max_admins:]
