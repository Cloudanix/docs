- This event is triggered when a user or an automated process initiates the deletion of an execution in CloudRun.
- The event provides information about the execution that was deleted, such as its ID, job name, and any associated metadata.

from google.cloud import run_v1

def delete_job_execution(project_id, location, service_name, job_name, execution_id):
    client = run_v1.CloudRunJobsServiceClient()
    parent = client.job_path(project_id, location, service_name, job_name)
    execution_path = f"{parent}/executions/{execution_id}"

    client.delete_execution(name=execution_path)
